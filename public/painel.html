// --- GESTÃO DE PEDIDOS --- function carregarPedidosRealTime() { db.collection('vendas_concluidas').orderBy('data_pedido', 'desc').limit(15) .onSnapshot(snap => { const tabela = document.getElementById('tabela-pedidos'); tabela.innerHTML = ''; snap.forEach(doc => { const p = doc.data(); // MUDANÇA AQUI: Pegando Data e Hora completas (Ex: 13/04/2026 20:30:15) const dataCompleta = p.data_pedido ? p.data_pedido.toDate().toLocaleString('pt-BR') : '--/--/---- --:--'; tabela.innerHTML += ` ${dataCompleta} ${p.cliente.nome} ${p.itens.map(i => i.nome).join(', ')} R$ ${p.financeiro.total_pago_reais.toFixed(2)} Pago `; }); }); }
